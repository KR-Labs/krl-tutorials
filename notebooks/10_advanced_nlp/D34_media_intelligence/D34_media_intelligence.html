def load_api_key(key_name: str) -> str:
    """
    Load API key from environment variable or .env file.
    
    Args:
        key_name: Name of the environment variable containing the API key
        
    Returns:
        API key string or None if not found
    """
    # Check environment variables
    api_key = os.getenv(key_name)
    
    if api_key:
        return api_key
    
    # Try loading from .env file in parent directories
    current_dir = Path.cwd()
    for parent in [current_dir] + list(current_dir.parents):
        env_file = parent / '.env'
        if env_file.exists():
            with open(env_file, 'r') as f:
                for line in f:
                    if line.strip() and not line.startswith('#'):
                        if '=' in line:
                            var_name, var_value = line.strip().split('=', 1)
                            if var_name == key_name:
                                return var_value.strip('"\' ')
    
    return None

# Text preprocessing pipeline for NLP analysis
class TextPreprocessor:
    """
    Text preprocessing pipeline for news article analysis.
    """
    
    def __init__(self, language='english'):
        self.language = language
        self.lemmatizer = WordNetLemmatizer()
        self.stop_words = set(stopwords.words(language))
        
        # Add custom stop words for news articles
        self.stop_words.update([
            'said', 'says', 'according', 'report', 'article', 'news',
            'told', 'new', 'also', 'would', 'could', 'may', 'update'
        ])
    
    def preprocess(self, text: str) -> str:
        """
        Clean and preprocess text.
        
        Args:
            text: Raw text string
            
        Returns:
            Cleaned and preprocessed text
        """
        if not isinstance(text, str):
            return ""
        
        # Convert to lowercase
        text = text.lower()
        
        # Tokenize
        tokens = word_tokenize(text)
        
        # Remove stopwords, lemmatize, and filter
        tokens = [
            self.lemmatizer.lemmatize(token)
            for token in tokens
            if token.isalpha() and 
               token not in self.stop_words and 
               len(token) > 3
        ]
        
        return ' '.join(tokens)
    
    def preprocess_corpus(self, texts: list) -> list:
        """
        Preprocess a corpus of texts.
        
        Args:
            texts: List of text strings
            
        Returns:
            List of preprocessed texts
        """
        return [self.preprocess(text) for text in texts]

)

# Extract top words for each topic
def get_top_words(model, feature_names, n_words=10):
    """Extract top words for each topic."""
    topics = []
    for topic_idx, topic in enumerate(model.components_):
        top_indices = topic.argsort()[-n_words:][::-1]
        top_words = [feature_names[i] for i in top_indices]
        topics.append(top_words)
    return topics

lda_topic_words = get_top_words(lda_model, feature_names, n_top_words)

# Assign dominant topic to each document

news_data['lda_topic_prob'] = lda_topics.max(axis=1)

    # Determine optimal number of clusters (cap at 8)
    n_clusters = min(8, max(3, len(geo_data) // 30))
    
    # Perform K-Means clustering
    kmeans = KMeans(n_clusters=n_clusters, random_state=RANDOM_SEED, n_init=10)
    geo_data['geo_cluster'] = kmeans.fit_predict(coords)
    
    # Calculate cluster centers
    cluster_centers = pd.DataFrame(
        kmeans.cluster_centers_,
        columns=['center_lat', 'center_lon']
    )
    cluster_centers['cluster'] = range(n_clusters)
    
    # Count articles per cluster
    cluster_counts = geo_data['geo_cluster'].value_counts().to_dict()
    cluster_centers['article_count'] = cluster_centers['cluster'].map(cluster_counts)
    
    print(f"\n Identified {n_clusters} geographic clusters")
    print(f"\nCluster Statistics:")
    print(cluster_centers)
    
    # Calculate average sentiment per cluster
    cluster_sentiment = geo_data.groupby('geo_cluster')['sentiment_compound'].mean()
    cluster_centers['avg_sentiment'] = cluster_centers['cluster'].map(cluster_sentiment)
    
    # Merge cluster assignments back to main dataframe
    news_data = news_data.merge(
        geo_data[['geo_cluster']],
        left_index=True,
        right_index=True,
        how='left'
    )
    
    GEO_CLUSTERING_SUCCESS = True
    
else:

Key Findings and Interpretation

Topic Analysis

The LDA topic modeling identified 8 distinct narrative themes in the news coverage:

  • Dominant Topics: Review the word clouds above to identify the most prevalent themes in the dataset
  • Topic Distribution: Examine the bar chart showing how many articles belong to each topic
  • Topic Evolution: Consider how topics might shift over time (visible in time series if dataset spans multiple days)

Sentiment Patterns

VADER sentiment analysis reveals the overall tone of media coverage:

  • Average Sentiment: Check whether coverage tends toward positive, neutral, or negative framing
  • Sentiment Variability: High standard deviation indicates controversial or polarizing topics
  • Temporal Trends: Look for sentiment shifts over time in the time series plot
  • Topic-Specific Sentiment: The heatmap shows which topics receive more positive vs. negative coverage

Geographic Insights

Geographic clustering reveals patterns in media attention:

  • Attention Clusters: Identify regions receiving concentrated news coverage
  • Regional Sentiment: Compare sentiment scores across different geographic clusters
  • Coverage Gaps: Note regions with sparse coverage despite potential relevance

Methodology Notes

  • LDA vs. BERTopic: LDA provides interpretable bag-of-words topics, while BERTopic captures semantic context
  • VADER Limitations: Works best for social media text; may need calibration for formal news
  • GDELT Coverage: Biases toward English-language and Western media sources
  • Time Window: Results are sensitive to the query period (7 days by default)
